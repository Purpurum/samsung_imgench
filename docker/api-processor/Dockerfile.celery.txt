FROM mundialis/esa-snap:latest

ENV PYTHONUNBUFFERED=1
ENV VIRTUAL_ENV=/opt/venv
ENV PATH="$VIRTUAL_ENV/bin:$PATH"
ENV PYTHONPATH=/app/src:/usr/local/snap/snappy:${PYTHONPATH}

RUN if command -v apk >/dev/null 2>&1; then \
        # Alpine Linux
        apk add --no-cache py3-pip; \
    elif command -v apt-get >/dev/null 2>&1; then \
        # Debian/Ubuntu
        apt-get update && apt-get install -y --no-install-recommends python3-pip && rm -rf /var/lib/apt/lists/*; \
    elif command -v yum >/dev/null 2>&1; then \
        # RHEL/CentOS
        yum install -y python3-pip && yum clean all; \
    else \
        # Fallback: bootstrap pip manually
        wget -q https://bootstrap.pypa.io/get-pip.py -O /tmp/get-pip.py && python3 /tmp/get-pip.py && rm /tmp/get-pip.py; \
    fi

RUN python3 -m pip install --no-cache-dir --break-system-packages \
    "celery==5.3.6" \
    "redis==5.0.1" \
    "boto3==1.34.34" \
    "shapely>=2.0.3" \
    "Pillow>=10.0.0"

WORKDIR /app
COPY app/ ./app/
